117class WKTElement(_SpatialElement): 

118 """ 

119 Instances of this class wrap a WKT or EWKT value. 

120 

121 Usage examples:: 

122 

123 wkt_element_1 = WKTElement('POINT(5 45)') 

124 wkt_element_2 = WKTElement('POINT(5 45)', srid=4326) 

125 wkt_element_3 = WKTElement('SRID=4326;POINT(5 45)', extended=True) 

126 """ 

127 

128 geom_from = 'ST_GeomFromText' 

129 geom_from_extended_version = 'ST_GeomFromEWKT' 

130 

131 def __init__(self, data, srid=-1, extended=False): 

132 if extended and srid == -1: 

133 # read srid from EWKT 

134 if not data.startswith('SRID='): 

135 raise ArgumentError('invalid EWKT string {}'.format(data)) 

136 data_s = data.split(';', 1) 

137 if len(data_s) != 2: 

138 raise ArgumentError('invalid EWKT string {}'.format(data)) 

139 header = data_s[0] 

140 try: 

141 srid = int(header[5:]) 

142 except ValueError: 

143 raise ArgumentError('invalid EWKT string {}'.format(data)) 

144 _SpatialElement.__init__(self, data, srid, extended)

158class WKBElement(_SpatialElement): 

159 """ 

160 Instances of this class wrap a WKB or EWKB value. 

161 

162 Geometry values read from the database are converted to instances of this 

163 type. In most cases you won't need to create ``WKBElement`` instances 

164 yourself. 

165 

166 If ``extended`` is ``True`` and ``srid`` is ``-1`` at construction time 

167 then the SRID will be read from the EWKB data. 

168 

169 Note: you can create ``WKBElement`` objects from Shapely geometries 

170 using the :func:`geoalchemy2.shape.from_shape` function. 

171 """ 

172 

173 geom_from = 'ST_GeomFromWKB' 

174 geom_from_extended_version = 'ST_GeomFromEWKB' 

175 

176 def __init__(self, data, srid=-1, extended=False): 

177 if extended and srid == -1: 

178 # read srid from the EWKB 

179 # 

180 # WKB struct { 

181 # byte byteOrder; 

182 # uint32 wkbType; 

183 # uint32 SRID; 

184 # struct geometry; 

185 # } 

186 # byteOrder enum { 

187 # WKB_XDR = 0, // Most Significant Byte First 

188 # WKB_NDR = 1, // Least Significant Byte First 

189 # } 

190 if isinstance(data, str_): 

191 # SpatiaLite case 

192 # assume that the string is an hex value 

193 header = binascii.unhexlify(data[:18]) 

194 else: 

195 header = data[:9] 

196 if not PY3: 

197 header = bytearray(header) 

198 byte_order, srid = header[0], header[5:] 

199 srid = struct.unpack('<I' if byte_order else '>I', srid)[0] 

200 _SpatialElement.__init__(self, data, srid, extended) 

201

203 def desc(self): 

204 """ 

205 This element's description string. 

206 """ 

207 if isinstance(self.data, str_): 

208 # SpatiaLite case 

209 return self.data 

210 desc = binascii.hexlify(self.data) 

211 if PY3: 

212 # hexlify returns a bytes object on py3 

213 desc = str(desc, encoding="utf-8") 

214 return desc
