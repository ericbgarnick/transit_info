74 def spy(self, obj, name): 

75 """ 

76 Creates a spy of method. It will run method normally, but it is now 

77 possible to use `mock` call features with it, like call count. 

78 

79 :param object obj: An object. 

80 :param unicode name: A method in object. 

81 :rtype: mock.MagicMock 

82 :return: Spy object. 

83 """ 

84 method = getattr(obj, name) 

85 

86 autospec = inspect.ismethod(method) or inspect.isfunction(method) 

87 # Can't use autospec classmethod or staticmethod objects 

88 # see: https://bugs.python.org/issue23078 

89 if inspect.isclass(obj): 

90 # Bypass class descriptor: 

91 # http://stackoverflow.com/questions/14187973/python3-check-if-method-is-static 

92 try: 

93 value = obj.__getattribute__(obj, name) 

94 except AttributeError: 

95 pass 

96 else: 

97 if isinstance(value, (classmethod, staticmethod)): 

98 autospec = False

100 def wrapper(*args, **kwargs): 

101 spy_obj.spy_return = None 

102 spy_obj.spy_exception = None 

103 try: 

104 r = method(*args, **kwargs) 

105 except Exception as e: 

106 spy_obj.spy_exception = e 

107 raise 

108 else: 

109 spy_obj.spy_return = r 

110 return r 

111 

112 async def async_wrapper(*args, **kwargs): 

113 spy_obj.spy_return = None 

114 spy_obj.spy_exception = None 

115 try: 

116 r = await method(*args, **kwargs) 

117 except Exception as e: 

118 spy_obj.spy_exception = e 

119 raise 

120 else: 

121 spy_obj.spy_return = r 

122 return r 

123 

124 if asyncio.iscoroutinefunction(method): 

125 wrapped = functools.update_wrapper(async_wrapper, method) 

126 else: 

127 wrapped = functools.update_wrapper(wrapper, method) 

128 

129 spy_obj = self.patch.object(obj, name, side_effect=wrapped, autospec=autospec) 

130 spy_obj.spy_return = None

145 class _Patcher: 

146 """ 

147 Object to provide the same interface as mock.patch, mock.patch.object, 

148 etc. We need this indirection to keep the same API of the mock package.

156 def _start_patch(self, mock_func, *args, **kwargs): 

157 """Patches something by calling the given function from the mock 

158 module, registering the patch to stop it later and returns the 

159 mock object resulting from the mock call. 

160 """ 

161 self._enforce_no_with_context(inspect.stack()) 

162 p = mock_func(*args, **kwargs) 

163 mocked = p.start() 

164 self._patches.append(p) 

165 if hasattr(mocked, "reset_mock"): 

166 self._mocks.append(mocked) 

167 return mocked

169 def _enforce_no_with_context(self, stack): 

170 """raises a ValueError if mocker is used in a with context""" 

171 caller = stack[2] 

172 frame = caller[0] 

173 info = inspect.getframeinfo(frame) 

174 if info.code_context is None: 

175 # no source code available (#169) 

176 return 

177 code_context = " ".join(info.code_context).strip() 

178 

179 if code_context.startswith("with mocker."): 

180 raise ValueError( 

181 "Using mocker in a with context is not supported. " 

182 "https://github.com/pytest-dev/pytest-mock#note-about-usage-as-context-manager" 

183 )
