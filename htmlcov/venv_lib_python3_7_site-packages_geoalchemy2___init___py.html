39 def dispatch(event, table, bind): 

40 if event in ('before-create', 'before-drop'): 

41 # Filter Geometry columns from the table with management=True 

42 # Note: Geography and PostGIS >= 2.0 don't need this 

43 gis_cols = [c for c in table.c if 

44 isinstance(c.type, Geometry) and 

45 c.type.management is True] 

46 

47 # Find all other columns that are not managed Geometries 

48 regular_cols = [x for x in table.c if x not in gis_cols] 

49 

50 # Save original table column list for later 

51 table.info["_saved_columns"] = table.c 

52 

53 # Temporarily patch a set of columns not including the 

54 # managed Geometry columns 

55 column_collection = expression.ColumnCollection() 

56 for col in regular_cols: 

57 column_collection.add(col) 

58 table.columns = column_collection 

59 

60 if event == 'before-drop': 

61 # Drop the managed Geometry columns 

62 for c in gis_cols: 

63 if bind.dialect.name == 'sqlite': 

64 drop_func = 'DiscardGeometryColumn' 

65 elif bind.dialect.name == 'postgresql': 

66 drop_func = 'DropGeometryColumn' 

67 else: 

68 raise ArgumentError('dialect {} is not supported'.format(bind.dialect.name)) 

69 args = [table.schema] if table.schema else [] 

70 args.extend([table.name, c.name]) 

71 

72 stmt = select([getattr(func, drop_func)(*args)]) 

73 stmt = stmt.execution_options(autocommit=True) 

74 bind.execute(stmt) 

75 

76 elif event == 'after-create': 

77 # Restore original column list including managed Geometry columns 

78 table.columns = table.info.pop('_saved_columns') 

79 

80 for c in table.c:

83 args = [table.schema] if table.schema else []
